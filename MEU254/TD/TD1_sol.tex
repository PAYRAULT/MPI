\subsection*{Rappel de cours}

\begin{defn}
La fonction indicatrice $1_A$ sur l'ensemble $A$ est d\'efinie comme:
$$
1_A : A \to \{0,1\}, x \to 1_A(x) = 
\left\{ 
\begin{array}{l}
0, \text{ si } x \notin A \\
1, \text{ si } x \in A \\
\end{array}
\right.
$$
\end{defn}

\subsection*{Exercice 1}
On a $1_{A^c} = 1 - 1_{A}$ car si $1_A(c) = 1$ alors $1_{A^c} = 0$ car quand $x \in A$ alors $x \notin A^c$ (m\^me raisonnement avec $1_A(x) = 0$), et $1_{A \cup B} = 1_A + 1_B - 1_A.1_B$ donc
$$1_{A^c \cup B^c} = 1_{A^c} + 1_{B^c} - 1_{A^c}.1_{B^c} = (1 - 1_{A}) + (1 - 1_{B}) - (1 - 1_{A}).(1 - 1_{B}) =  1 - 1_{A} + 1 - 1_{B} - (1 - 1_{A} - 1_{B} + 1_{A}.1_{B}) = 1 - 1_{A}.1_{B}$$ 


QED
